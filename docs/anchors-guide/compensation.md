def calculate_anchor_reward(milestone_value, complexity, performance):
    # Base calculation
    base_reward = milestone_value * 0.03  # 3% base rate
    
    # Complexity multiplier
    complexity_bonus = {
        "simple": 1.0,
        "moderate": 1.3,

    return total_reward
